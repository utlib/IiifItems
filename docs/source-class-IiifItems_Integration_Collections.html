116:         $record = $args['record'];
117:         // Unset sensitive fields
118:         if (isset($args['post']['Elements'][$uuidElementId])) {
119:             unset($args['post']['Elements'][$uuidElementId]);
120:         }
121:         // Add UUID if it's new
122:         if ($args['insert']) {
123:             $record->addTextForElement($uuidElement, generate_uuid());
124:         }
125:         // Check Parent Collection element text
126:         if (isset($args['post']['Elements'][$parentElementId])) {
127:             $parentUuid = $args['post']['Elements'][$parentElementId][0]['text'];
128:             if ($parentUuid) {
129:                 // Manifests can't be parents
130:                 $parent = find_collection_by_uuid($parentUuid);
131:                 if (raw_iiif_metadata($parent, 'iiifitems_collection_type_element') != 'Collection') {
132:                     $record->addError('Parent Collection', __('A collection can only have collection-type collections as its parent.'));
133:                 }
134:                 // Anti-loop check if is collection has a parent
135:                 $visitedUuids = array($record->getElementTextsByRecord($uuidElement)[0]->text => true);
136:                 $current = $parent;
137:                 while ($current) {
138:                     $currentUuid = raw_iiif_metadata($current, 'iiifitems_collection_uuid_element');
139:                     if (isset($visitedUuids[$currentUuid])) {
140:                         $record->addError('Parent Collection', __('A collection cannot have itself or a descendent as its parent.'));
141:                         break;
142:                     }
143:                     $visitedUuids[$currentUuid] = true;
144:                     $current = find_collection_by_uuid(raw_iiif_metadata($current, 'iiifitems_collection_parent_element'));
