118:     /**
119:      * Attempts to return the manifest URL from IIIF drag-and-drop URLs
120:      * @param string $url
121:      * @return string
122:      */
123:     protected function _extractResourceUrl($url) {
124:         // Return URL as is if there are no queries
125:         $queryPos = strpos($url, '?');
126:         if ($queryPos === false) {
127:             return $url;
128:         }
129:         // Parse the query portion (i.e. after ?) and return any manifest found
130:         $querySection = substr($url, $queryPos+1);
131:         parse_str($querySection, $parsed);
132:         if (isset($parsed['manifest'])) {
133:             return $parsed['manifest'];
134:         }
135:         // Still nothing, just give back the URL
136:         return $url;
137:     }
