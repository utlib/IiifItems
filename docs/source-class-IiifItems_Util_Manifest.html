 47:      * Return the IIIF Presentation API manifest representation of the Omeka collection
 48:      * @param Collection $collection
 49:      * @param boolean $bare Whether to exclude the annotation list references
 50:      * @return array
 51:      */
 52:     public static function buildManifest($collection, $bare=false) {
 53:         // Set default IDs and titles
 54:         $atId = public_full_url(array('things' => 'collections', 'id' => $collection->id, 'typeext' => 'manifest.json'), 'iiifitems_oa_uri');
 55:         $seqId = public_full_url(array('things' => 'collections', 'id' => $collection->id, 'typeext' => 'sequence.json'), 'iiifitems_oa_uri');
 56:         $label = metadata($collection, array('Dublin Core', 'Title'), array('no_escape' => true));
 57:         // Do it only for manifests with appropriate authorization
 58:         if (self::isManifest($collection)) {
 59:             // Decide which cache entry to consider
 60:             $cacheEntryName = $bare ? 'private_bare_manifest' : (
 61:                 current_user() ? 'private_manifest' : 'public_manifest'
 62:             );
 63:             if ($json = get_cached_iiifitems_value_for($collection, $cacheEntryName)) {
 64:                 return $json;
 65:             }
 66:             // Try to find template; if it does not already exist, use the blank template
 67:             if (!($json = parent::fetchJsonData($collection))) {
 68:                 $json = self::blankTemplate($atId, $seqId, $label);
 69:             }
 70:             // Override the IDs, titles and DC metadata
 71:             $json['@id'] = $atId;
 72:             $json['sequences'][0]['@id'] = $seqId;
 73:             $json['sequences'][0]['canvases'] = self::findCanvasesFor($collection);
 74:             parent::addDublinCoreMetadata($json, $collection);
 75:             // Cache accordingly
 76:             cache_iiifitems_value_for($collection, $json, $cacheEntryName);
 77:             // Done
 78:             return $json;
 79:         }
 80:         return self::blankTemplate($atId, $seqId, $label);
 81:     }
