271:     /**
272:      * Return the multimedia type of the files given.
273:      * @param array $files
274:      * @return string "audio", "file", "image", "pdf", "text" or "video"
275:      */
276:     protected static function _getNonIiifType($files) {
277:         $candidateType = null;
278:         // Go through the list of files
279:         foreach ($files as $file) {
280:             // Sniff mime types
281:             $currentMime = $file->mime_type;
282:             if ($currentMime == 'application/pdf') {
283:                 $currentType = 'pdf';
284:             }
285:             elseif ($currentMime == 'application/zip' || strpos($currentMime, 'compressed') !== false || $currentMime == 'application/x-gtar' || $currentMime == 'application/x-tar' || $currentMime == 'application/gzip') {
286:                 $currentType = 'zip';
287:             }
288:             else {
289:                 switch ($currentMimePrefix = substr($currentMime, 0, 5)) {
290:                     case 'audio': case 'image': case 'video':
291:                         $currentType = $currentMimePrefix;
292:                     break;
293:                     case 'text/':
294:                         $currentType = 'text';
295:                     break;
296:                     default:
297:                         $currentType = 'file';
298:                     break;
299:                 }
300:             }
301:             // First file should determine the starting type
302:             if ($candidateType === null) {
303:                 $candidateType = $currentType;
304:             }
305:             // Otherwise, if the file doesn't match the first file's type, return mixed
306:             else if ($candidateType != $currentType) {
307:                 return 'mixed';
308:             }
309:         }
310:         return ($candidateType === null) ? 'none' : $candidateType;
311:     }
